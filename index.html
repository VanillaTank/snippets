  class Circle {
    constructor(xPoint, yPoint, radius, color) {
      this.xPoint = xPoint;
      this.yPoint = yPoint;
      this.radius = radius;
      this.color = color;
    }

    draw(ctx) {
      ctx.beginPath();
      ctx.arc(this.xPoint, this.yPoint, this.radius, 0, Math.PI * 2, false)
      ctx.strokeStyle = 'grey';
      ctx.lineWidth = 1;
      ctx.fillStyle = this.color;
      ctx.fill();
      ctx.stroke();
      ctx.closePath();
    }

    changeColor(newColor) {
      this.color = newColor;
      this.draw(ctx);
    }

    clickCircle(xMouse, yMouse) {
      const distance =
        Math.sqrt(
          ((xMouse - this.xPoint) * (xMouse - this.xPoint))
          +
          ((yMouse - this.yPoint) * (yMouse - this.yPoint))
        );

      if (distance <= this.radius) {
        this.changeColor('#56f')
        return true;
      } else {
        this.changeColor('#f56')
        return false;
      }

    }
  }

  let circle = new Circle(150, 150, 50, '#f56');
  circle.draw(ctx)

  canvas.addEventListener('click', (event) => {
    const rect = canvas.getBoundingClientRect();
    const x = event.clientX - rect.left;
    const y = event.clientY - rect.top;
    circle.clickCircle(x, y)
  })

    const DEBOUNCE_INTERVAL = 300;
    function debounce(fun) {
        let lastTimeout = null;

        return function () {
            const args = arguments;
            if (lastTimeout) {
                console.log(1);
                window.clearTimeout(lastTimeout)
            }
            lastTimeout = window.setTimeout(function () {
                lastTimeout = null;
                fun.apply(null, args)
            }, DEBOUNCE_INTERVAL)
        }
    }

      function insertMark(string, pos, len) {

        //len - количество символов, которые ввел пользователь
        return string.slice(0, pos) + "<mark>" + string.slice(pos, pos + len) + "</mark>" + string.slice(pos + len);
      }

    Подключение font awesome

    ко ко ко

    Подключение по ссылке

    1. Регаем свою почту на https://fontawesome.com/

    2. Получаем ссылку
      <script src="https://kit.fontawesome.com/682a1bab79.js" crossorigin="anonymous"></script>

    3. Вставляем класс или i куда нужно
      <a class="fas fa-hamsa" href="#"><i class="fas fa-hamsa"></i> ко ко ко</a>

    4. Внимание!!! При таком подключении настройки иконки(например, шрифт) будут влиять на наши стили, что может мешать.

    Локальное использование

    1. Скачиваем архив с официального сайта (будет лежать в на жестком диске)

    2. Заходим в папку css. All.css подключит все иконки, brand.css - брендовые (ех. Ютуб), и т.д. Внутренности можно посомтреть на сайте. Копируем нужный файл в свой проект и подключаем в link.

    3. Перетаскиваем в корень нашего проета папку со шрифтами webfonts. Если ее переименовать или взять из нее конкретный файл, в CSS нужно переписать url в самом конце файла.

    4. Вставляем класс или i куда нужно
      <a class="fas fa-hamsa" href="#"><i class="fas fa-hamsa"></i> ко ко ко</a>

    5. Чтобы пофиксить влияние стилей иконок на свои стили, нужно в самом низу их css прописать, что стили относятся к псевдоэлементам. Например, 

      .fa,
.fas::before{
  font-family: 'Font Awesome 5 Free';
  font-weight: 900; }

    6. Кстати, в папке svg лежат все иконки, можно таскать оттуда :)

    Как получить платную иконку бесплатно

    1. Заходим на сайт, выбираем нужную иконку

    2. Тыкаем "Посмотреть код" в ДевТулс. ПКМ, copy.

    3. Создаем локально svg-файл и сохранияем скопированное в него.
